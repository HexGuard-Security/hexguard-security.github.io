Lab 5 MAVLink Scripting

parse mavlink pcap export csv plot flight path

objective

extract global position messages save csv and plot path using matplotlib

script mav2csv py

from pymavlink import mavutil
import csv

m = mavutil.mavlink_connection('sitl_mav.pcap', dialect='ardupilotmega')
with open('positions.csv','w',newline='') as fh:
    w = csv.writer(fh)
    w.writerow(['time_boot_ms','lat','lon','alt'])
    while True:
        msg = m.recv_match(blocking=False)
        if msg is None:
            break
        if msg.get_type() == 'GLOBAL_POSITION_INT':
            w.writerow([msg.time_boot_ms, msg.lat/1e7, msg.lon/1e7, msg.alt/1000.0])

run

python3 mav2csv.py
python3 -c "import matplotlib.pyplot as plt; import csv; lat=[]; lon=[]; 
with open('positions.csv') as f: 
  r=csv.DictReader(f); 
  for row in r: 
    lat.append(float(row['lat'])); lon.append(float(row['lon']));
plt.plot(lon,lat); plt.xlabel('lon'); plt.ylabel('lat'); plt.savefig('flight_path.png')"
