Lab 6 Passive Tracker

compute spectrum peaks from iq and export rssi timeline

objective

detect signal activity in fpv band and produce csv of freq rssi time

script tracker py

import numpy as np
import struct
import argparse
import csv
from scipy.signal import welch

parser = argparse.ArgumentParser()
parser.add_argument('--input', required=True)
parser.add_argument('--output', required=True)
args = parser.parse_args()

samp_rate = 2000000
window = 1024
fmt = 'f'  # float32

def file_iter(fname, block_size):
    with open(fname,'rb') as f:
        while True:
            data = f.read(block_size*4*2)
            if not data:
                break
            iq = np.frombuffer(data, dtype=np.float32).astype(np.complex64)
            iq = iq[::2] + 1j*iq[1::2]
            yield iq

with open(args.output,'w',newline='') as fh:
    w = csv.writer(fh)
    w.writerow(['time_s','freq_hz','rssi_db'])
    t = 0.0
    dt = window / samp_rate
    for iq in file_iter(args.input, window):
        f, pxx = welch(iq, fs=samp_rate, nperseg=window)
        peak_idx = np.argmax(pxx)
        peak_freq = f[peak_idx]
        peak_rssi = 10 * np.log10(pxx[peak_idx])
        w.writerow([t, peak_freq, peak_rssi])
        t += dt
